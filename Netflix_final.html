import gradio as gr
import pandas as pd

# Load dataset
df = pd.read_csv("C:\\Users\\DELL\\Downloads\\netflix_titles.csv")

# Make sure these columns exist
df['release_year'] = pd.to_numeric(df['release_year'], errors='coerce')
df['main_country'] = df['country'].str.split(',').str[0]

# Filtering Function
def filter_data(type_filter, country_filter, year_filter):
    filtered = df[
        (df['type'] == type_filter) &
        (df['main_country'] == country_filter) &
        (df['release_year'] == year_filter)
    ]
    return filtered[['title', 'type', 'main_country', 'release_year']]

with gr.Blocks() as demo:
    gr.Markdown("#  Netflix Dashboard")

    with gr.Row():

        # LEFT SIDE (Filters Panel)
        with gr.Column(scale=1):
            gr.Markdown("##  Filter Options")

            type_filter = gr.Dropdown(
                choices=df['type'].dropna().unique().tolist(),
                label="Select Type"
            )

            country_filter = gr.Dropdown(
                choices=sorted(df['main_country'].dropna().unique().tolist()),
                label="Select Country"
            )

            year_filter = gr.Slider(
                minimum=int(df['release_year'].dropna().min()),
                maximum=int(df['release_year'].dropna().max()),
                step=1,
                label="Select Release Year"
            )

            filter_btn = gr.Button("Apply Filters")

        # RIGHT SIDE (Output Panel)
        with gr.Column(scale=2):
            output_table = gr.Dataframe()

        filter_btn.click(
            fn=filter_data,
            inputs=[type_filter, country_filter, year_filter],
            outputs=output_table
        )

demo.launch()
